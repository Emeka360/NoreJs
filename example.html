1.HTML text with Nore-js

      
<!DOCTYPE html>
<html>
 <head>
  <title>Write with NoreJS</title>
</head>
<body>
  
</body>
  <script src="nore.js"></script>
  <script>
     //The $('body') calls out the HTML Body.
    //The html is what writes the text into HTML Dom
    $('body').html('Hello NoreJS');
  </script>
</html>

2.FadeOut with NoreJS

        
<!DOCTYPE html>
<html>
  <head>
    <title>FadeOut With NoreJS</title>
  </head>
<body>
  
</body>
  <script src="nore.js"></script>
  <script>
    $('body').text('Hello NoreJS');
    $('body').on('click',function(){
      $('body').fadeOut(1000,'1s')
    });
  </script>
</html>

3.FadeIn with NoreJS

        
  <!DOCTYPE html>
  <html>
   <head>
  <title>FadeOut With NoreJS</title>
</head>
<body>
  <div>
     <p></p>
    <button>Click me</button>
  </div>
</body>
  <script src="nore.js"></script>
  <script>
    $('p').text('Hello NoreJS');
    $('p').hide();
    $('button').on('click',function(){
      $('p').fadeIn(1,'12s')
    });
  </script>
</html>
